// MessageList.jsx
import React from 'react';

function MessageList({ messages, onReplyClick }) {
    return (
        <div>
        {messages.map((message) => (
            <div key={message.id}>
            <p>{message.content}</p>
            <button onClick={() => onReplyClick(message.id)}>Reply</button>
            </div>
        ))}

        </div>
    );
}

export default MessageList;
